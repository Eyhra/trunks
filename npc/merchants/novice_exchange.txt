function	script	F_PotExchange	{
	.@item_req = getarg(0);
	.@req_amount = getarg(1);

	if (countitem(.@item_req) < .@req_amount) {
		mes "[Mercador]";
		mes "Ei, onde estão os "+ getitemname(.@item_req) +" que você prometeu?";
		mes "Me de "+ getitemname(.@item_req) +"!";
		close;
	}
	else {
		mes "[Mercador]";
		mes "Ok, deixe-me ver";
		mes "quantos "+ getitemname(.@item_req) +" você tem com você. Hmm...";
		next;
		mes "[Mercador]";
		mes "Você tem um total de "+ countitem(.@item_req) +" "+ getitemname(.@item_req) +".";
		mes "Eu posso te dar o total de "+ (countitem(.@item_req)/.@req_amount) +" Poções Vermelhas por isso.";
		next;
		mes "[Mercador]";
		mes "O que me diz?";
		mes "Negócio fechado?";
		next;
		if (select("Fechado.:Nada feito.") == 1) {
			mes "[Mercador]";
			mes "Você sabe exatamente o número de Poções Vermelhas que irá receber por esse "+ getitemname(.@item_req) +", não sabe?";
			next;
			mes "[Mercador]";
			mes "Agora, posso trocar com você no mínimo 1 Poção Vermelha e no máximo 100 de cada vez.";
			mes "Se você mudar de idéia, basta digitar '0' para cancelar.";
			next;
			while(1) {
				input .@input,0,101;
				if (.@input == 0) {
					mes "[Mercador]";
					mes "O que...?";
					mes "O que você pensa que está dizendo?";
					mes "Muito indeciso você, não?";
					close;
				}
				else if (.@input > 100) {
					mes "[Mercador]";
					mes "Eu não posso dar mais que 100 Poções Vermelhas de uma vez.";
					mes "Vamos começar de novo.";
					next;
				}
				else {
					break;
				}
			}
			.@put_out_item = .@input * .@req_amount;
			if (countitem(.@item_req) < .@put_out_item) {
				mes "[Mercador]";
				mes "Uh oh, o número que você digitou não parece certo. Olhe novamente o número de Poções Vermelhas que você pode trocar.";
				close;
			}
			mes "[Mercador]";
			mes "Onde você vai!";
			mes "Veja mais quantas Poções Vermelhas eu posso te dar.";
			mes "Obrigado, foi um bom negócio.";
			delitem .@item_req,.@put_out_item; //Venda
			getitem 501,.@input; //Poção Vermelha
			close;
		}
		mes "[Mercador]";
		mes "Hmmm~";
		mes "Tudo bem.";
		mes "Contudo não acha que Poções Vermelhas serão úteis para um aventureiro como você?";
		close;
	}
}
