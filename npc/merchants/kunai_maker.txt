function	script	Kunai_Trade	{
	setarray .@item_req[0], getarg(0), getarg(2);
	setarray .@req_amount[0], getarg(1), getarg(3);
	.@item_id = getarg(4);

	mes "[Kashin]";
	mes "Você pode trocar " + .@req_amount[0] + " " + getitemname(.@item_req[0]) + " e " + .@req_amount[1] + " " + getitemname(.@item_req[1]) + " por 10 " + getitemname(.@item_id) +".";
	next;
	mes "[Kashin]";
	mes "Eu posso só lhe dar de cada vez um máximo de 500 Kunais.";
	mes "Se quiser cancelar a troca basta que digite 0.";
	mes "Quantas Kunais você gostaria?";
	next;
	input .@amount;
	if(.@amount < 1 || .@amount > 500) {
		mes "[Kashin]";
		mes "Ahn? Eu sinto muito, mas eu não o posso dar uma quantidade acima de 500 unidades.";
		close;
	}
	if (countitem(.@item_req[0]) < .@amount*.@req_amount[0] || countitem(.@item_req[1]) < .@amount*.@req_amount[1]) {
		mes "[Kashin]";
		mes "Parece que você não tem a quantidade de itens necessários para a troca.";
		mes "Por favor, verifique seus itens e tente novamente.";
		close;
	}
	if (checkweight(.@item_id, .@amount*10) == 0) {
		mes "[Kashin]";
		mes "Parece que não vai poder carregar todas os itens.";
		mes "Por favor, verifique a quantidade de peso que carrega e tente novamente.";
		close;
	}
	mes "[Kashin]";
	mes "Ótimo, tudo está em ordem, assim prossigamos com a troca.";
	mes "Espero que faça bom proveito com suas novas Kunais.";
	delitem .@item_req[0],.@req_amount[0]*.@amount;
	delitem .@item_req[1],.@req_amount[1]*.@amount;
	getitem .@item_id,10*.@amount;
	close;
}
