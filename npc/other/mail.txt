// Floating mailbox npc that all NPCs duplicate from.
//============================================================
-	script	Caixa de Correio#dummy::MailBox	-1,{
	mes "[Caixa de Correio]";
	mes "Para usar o serviço de caixa de";
	mes "correio, você é obrigado a pagar";
	mes "130 zeny. Você gostaria de usar o serviço?";
	next;
	switch(select("Sim.:Não.")) {
	case 1:
		mes "[Caixa de Correio]";
		if (Zeny < 130) {
			mes "Desculpe, mas você não tem";
			mes "dinheiro suficiente. Para usar o";
			mes "serviço de caixa de correio, você é obrigado a pagar 130 zeny.";
			close;
		}
		mes "Obrigado, por favor volte novamente.";
		set Zeny, Zeny - 130;
		close2;
		openmail;
		end;
	case 2:
		mes "[Caixa de Correio]";
		mes "Obrigado, por favor volte novamente.";
		close;
	}
}
