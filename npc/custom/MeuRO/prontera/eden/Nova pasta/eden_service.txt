-	script	Goods Cabinet#00::pggc	-1,{
	mes "É um armário de armazenamento de mercadorias.";
	mes "Uma mensagem é escrita em um pedaço de papel.";
	next;
	mes "+ Gabinete é exclusivamente +";
	mes "+ para o grupo Éden +";
	mes "Se você quiser usar este";
	mes "gabinete por favor verifique";
	mes "a seguir:";
	next;
	mes "1.Você é um membro";
	mes "   do grupo Éden?";
	mes "2.Você aprendeu";
	mes "   habilidades básicas suficientes?";
	mes "3.A taxa de gabinete é";
	mes "  ^4d4dff500 zeny^000000!";
	next;
	if (callfunc("F_HasEdenGroupMark")) {
		mes "Você precisa inserir zeny para usar o gabinete.";
		mes "Custo : 500 Zeny ";
		mes "Você gostaria de usá-lo?";
		next;
		switch(select("^3CB371[»]^000000 Use o Gabinete:^ff0000[»]^000000 Cancelar")) {
			case 1:
				if (Zeny > 499) {
					if(!callfunc("F_CanOpenStorage")){
						mes "O gabinete não está funcionando para mim.";
						mes "Talvez eu ainda não esteja qualificado para usar o Gabinete.";
						close;
					}
					set Zeny, Zeny - 500;
					close2;
					openstorage;
					end;
				}
				mes "Eu não tenho zeny suficiente.";
				mes "Preciso de pelo menos 500 zeny para usar o Gabinete.";
				close;
			case 2:
				mes "Vou usar da próxima vez.";
				close;
		}
	}
	mes "Eu preciso de uma Marca do Grupo Éden para usar este Gabinete.";
	close;
}
