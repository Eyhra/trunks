-	script	hold	-1,{

	OnInit:
		bindatcmd("hold", strnpcinfo(3)+"::OnHold");
		end;

	OnHold:
		if ( !@Hold ) {
			set @Hold, 1;
			message strcharinfo(0), "Hold está ativo.";
			pcblockmove getcharid(3), 1;
		} else {
			set @Hold, 0;
			message strcharinfo(0), "Hold desativado.";
			pcblockmove getcharid(3), 0;
		}

}
